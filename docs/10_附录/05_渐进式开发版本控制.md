func _generate_test_terrain() -> void:
	# 铺满草地 (gz=0)
	var grass: BlockData = BlockData.create("grass")
	block_grid.fill_rect(0, 0, TEST_AREA_SIZE - 1, TEST_AREA_SIZE - 1, 0, grass)

	# 石头障碍（不可通行）
	var stone: BlockData = BlockData.create("stone", 0, 2)
	block_grid.set_block(8, 8, 0, stone)
	block_grid.set_block(9, 8, 0, stone)
	block_grid.set_block(10, 8, 0, stone)
	block_grid.set_block(10, 9, 0, stone)
	block_grid.set_block(10, 10, 0, stone)

	# 灵田方块（可交互）
	var farm: BlockData = BlockData.create("farm")
	for gx: int in range(12, 16):
		for gy: int in range(12, 16):
			block_grid.set_block(gx, gy, 0, farm)
		}
	}

	# 水域
	var water: BlockData = BlockData.create("water")
	block_grid.set_block(0, 10, 0, water)
	block_grid.set_block(0, 11, 0, water)
	block_grid.set_block(1, 10, 0, water)

	# 强制重建渲染（因为 block_renderer 可能在 grid 设置时还没 ready）
	if block_renderer:
		block_renderer.rebuild_all()
	}
	if block_collision:
		block_collision.build_collision_from_grid()
	}

func _connect_signals() -> void:
	# 方块交互 → 灵田种植/收获
	SignalBus.block_interacted.connect(_on_block_interacted)
	# 日更新 → 作物生长
	SignalBus.day_changed.connect(_on_day_changed)
	# 敌人死亡 → 掉落
	SignalBus.enemy_killed.connect(_on_enemy_killed)

func _on_block_interacted(gx: int, gy: int, _gz: int, _player_id: int) -> void:
	var block: BlockData = block_grid.get_block(gx, gy, 0)
	if block and block.block_type == "farm":
		# 检查是否有成熟作物可收获
		var harvest: Dictionary = field_system.harvest(gx, gy)
		if harvest.is_empty():
			# 空灵田 → 种植灵稻
			field_system.plant(gx, gy, "spirit_rice")
			print("[WorldController] Planted spirit_rice at (%d,%d)" % [gx, gy])
		else:
			print("[WorldController] Harvested %s x%d quality:%s" % [harvest["crop_type"], harvest["yield"], harvest["quality"]])
		}
	}

func _on_day_changed(_day: int, _season: String) -> void:
	field_system.on_day_passed()
	print("[WorldController] Day %d, season: %s" % [_day, _season])

## EconomySystem — 经济与货币系统
## 管理灵石/灵玉两种货币，提供增减与查询接口。

enum Currency {
	SPIRIT_STONE = 0,  # 灵石（基础货币）
	SPIRIT_JADE = 1,    # 灵玉（稀有货币）
}
const JADE_TO_STONE_RATE: int = 100  # 1灵玉 = 100灵石

signal currency_changed(currency: int, new_amount: int)

var _spirit_stones: int = 0
var _spirit_jade: int = 0

func add_currency(type: int, amount: int) -> void:
	match type:
		Currency.SPIRIT_STONE:
			_spirit_stones = maxi(0, _spirit_stones + amount)
			currency_changed.emit(type, _spirit_stones)
		}Currency.SPIRIT_JADE:
			_spirit_jade = maxi(0, _spirit_jade + amount)
			currency_changed.emit(type, _spirit_jade)
		}
	}

func spend(type: int, amount: int) -> bool:
	match type:
		Currency.SPIRIT_STONE:
			if _spirit_stones < amount:
				return false
			}
			add_currency(type, -amount)
		}Currency.SPIRIT_JADE:
			if _spirit_jade < amount:
				return false
			}
			add_currency(type, -amount)
		}
	}
	return true

func get_stones() -> int:
	return _spirit_stones
func get_jade() -> int:
	return _spirit_jade

func collect_seal(seal_id: String) -> void:
	if seal_id in _collected_seals:
		return
	}
	_collected_seals.append(seal_id)
	seal_unlocked.emit(seal_id)
	SignalBus.seal_collected.emit(seal_id)

func equip_seal(seal_id: String, slot: int) -> bool:
	if slot < 0 or slot >= MAX_SEAL_SLOTS:
		return false
	}
	if not seal_id in _collected_seals:
		return false
	}
	while _equipped_seals.size() <= slot:
		_equipped_seals.append("")
	}
	_equipped_seals[slot] = seal_id
	SignalBus.seal_equipped.emit(seal_id, slot)
	_check_set_activation()
	return true

func unequip_seal(slot: int) -> String:
	if slot < 0 or slot >= _equipped_seals.size():
		return ""
	}
	var seal_id: String = _equipped_seals[slot]
	_equipped_seals[slot] = ""
	SignalBus.seal_unequipped.emit(seal_id, slot)
	return seal_id

func _check_set_activation() -> void:
	for set_id: String in _seal_sets:
		var members: Array = _seal_sets[set_id]["members"]
		var equipped_count: int = 0
		for seal: String in members:
			if seal in _equipped_seals:
				equipped_count += 1
			}
		}
		if equipped_count >= 2:
			set_activated.emit(set_id)
			SignalBus.seal_set_activated.emit(set_id)
		}
	}

func add_quest(id: String, type: int, title: String, objectives: Array, rewards: Dictionary = {{}}, prerequisites: Array = []) -> void:
	_quests[id] = {"type": type, "title": title, "objectives": objectives, "rewards": rewards, "prerequisites": prerequisites, "progress": [], "state": "locked"}

func start_quest(quest_id: String) -> bool:
	var q: Dictionary = _quests.get(quest_id, {{}})
	if q.is_empty() or q["state"] != "locked":
		return false
	}
	q["state"] = "active"
	_active_quests.append(quest_id)
	SignalBus.quest_started.emit(quest_id)
	return true

func progress_objective(quest_id: String, objective_id: String, amount: int = 1) -> void:
	var q: Dictionary = _quests.get(quest_id, {{}})
	if q.is_empty() or q["state"] != "active":
		return
	}
	var progress: Dictionary = q["progress"]
	progress[objective_id] = progress.get(objective_id, 0) + amount
	SignalBus.quest_progressed.emit(quest_id, objective_id)
	if _check_quest_complete(quest_id):
		_complete_quest(quest_id)
	}

func _check_quest_complete(quest_id: String) -> bool:
	var q: Dictionary = _quests[quest_id]
	var progress: Dictionary = q["progress"]
	for obj: Dictionary in q["objectives"]:
		var required: int = obj.get("count", 1)
		if progress.get(obj["id"], 0) < required:
			return false
		}
	}
	return true

func _complete_quest(quest_id: String) -> void:
	_quests[quest_id]["state"] = "completed"
	_active_quests.erase(quest_id)
	_completed_quests.append(quest_id)
	SignalBus.quest_completed.emit(quest_id)
	# 发放奖励
	for item: String in _quests[quest_id]["rewards"]:
		SignalBus.item_collected.emit(item, _quests[quest_id]["rewards"][item])
	}


signal save_completed(slot: int)

func save_game(slot: int) -> bool:
	if slot < 0 or slot >= MAX_SLOTS:
		return false
	}
	DirAccess.make_dir_recursive_absolute(SAVE_DIR)
	var data: Dictionary = _collect_save_data()
	var json: String = JSON.stringify(data, "")
	var path: String = SAVE_DIR + "save_%d.json" % slot
	var file: FileAccess = FileAccess.open(path, FileAccess.WRITE)
	if not file:
		return false
	}
	file.store_string(json)
	file.close()
	save_completed.emit(slot)
	SignalBus.game_loaded.emit(slot)  # 通过 GameManager 的 game_saved
	return true

func load_game(slot: int) -> bool:
	if slot < 0 or slot >= MAX_SLOTS:
		return false
	}
	var path: String = SAVE_DIR + "save_%d.json" % slot
	if not FileAccess.file_exists(path):
		return false
	}
	var file: FileAccess = FileAccess.open(path, FileAccess.READ)
	var json: String = file.get_as_text()
	file.close()
	var data: Dictionary = JSON.parse_string(json) as Dictionary
	if data.is_empty() or not _check_version(data):
		return false
	}
	_apply_save_data(data)
	load_completed.emit(slot)
	return true

func delete_save(slot: int) -> void:
	var path: String = SAVE_DIR + "save_%d.json" % slot
	if FileAccess.file_exists(path):
		DirAccess.remove_absolute(path)
	}

func _collect_save_data() -> Dictionary:

	GameManager.world_data = data.get("world_data", {{}})
	GameManager.game_time = data.get("game_time", {"day":1,"hour":6,"minute":0,"season":"spring"})

func _check_version(data: Dictionary) -> bool:
	var ver: String = data.get("version", "")

const REALM_EXP_REQUIREMENTS: Array[int] = [0, 100, 300, 800, 2000, 5000, 12000, 30000, 80000]
const REALM_COEFF_BASE: float = 0.15

signal realm_changed(old_realm: int, new_realm: int)
signal exp_gained(amount: int)

var current_realm: int = 1
var current_exp: int = 0
var total_exp: int = 0

func add_exp(amount: int) -> void:
	if current_realm >= MAX_REALM:
		return
	}
	current_exp += amount
	total_exp += amount
	exp_gained.emit(amount)
	while current_realm < MAX_REALM and current_exp >= REALM_EXP_REQUIREMENTS[current_realm]:
		current_exp -= REALM_EXP_REQUIREMENTS[current_realm]
		var old: int = current_realm
		current_realm += 1
		realm_changed.emit(old, current_realm)
		SignalBus.player_realm_changed.emit(old, current_realm)
	}

func get_realm_name() -> String:
	return REALM_NAMES[current_realm - 1]

func get_exp_to_next() -> int:
	if current_realm >= MAX_REALM:
		return 0
	}
	return REALM_EXP_REQUIREMENTS[current_realm]

func get_realm_coeff(atk_realm: int, def_realm: int) -> float:
	return maxf(1.0 + (atk_realm - def_realm) * REALM_COEFF_BASE, 0.3)

		var tree: SceneTree = get_tree()

	if dist <= attack_range:

	}

## 计算最终伤害
## [param attack] 攻击方攻击力
## [param defense] 防御方防御力
## [param atk_element] 攻击元素
## [param def_element] 防御元素
## [param atk_realm] 攻击方境界 (1-9)
## [param def_realm] 防御方境界 (1-9)
## [param skill_multiplier] 技能倍率 (默认 1.0)
func calculate_damage(attack: int, defense: int, atk_element: int, def_element: int, atk_realm: int = 1, def_realm: int = 1, skill_multiplier: float = 1.0) -> int:
	# 1. 五行克制倍率
	var element_mult: float = _get_element_multiplier(atk_element, def_element)

	# 2. 境界压制系数
	var realm_coeff: float = 1.0 + (atk_realm - def_realm) * REALM_COEFF_BASE
	realm_coeff = maxf(realm_coeff, 0.3)  # 最低30%，避免完全免疫

	# 3. 计算公式
	var raw_damage: float = attack * element_mult * realm_coeff * skill_multiplier
	var final_damage: int = maxi(1, int(raw_damage - defense * 0.5))
	return final_damage

## 获取五行克制倍率
func _get_element_multiplier(atk: int, def: int) -> float:
	if atk == Element.NONE or def == Element.NONE:
		return SAME_MULTIPLIER
	}
	if ELEMENT_COUNTER.get(atk, -1) == def:
		return COUNTER_MULTIPLIER  # 克制：150%
	}
	if ELEMENT_COUNTER.get(def, -1) == atk:
		return RESIST_MULTIPLIER  # 被克：70%
	}
	return SAME_MULTIPLIER

func melee_hitbox(origin: Vector2, direction: Vector2, range: float, mask: int = 2) -> Array:
	var space: PhysicsDirectSpaceState2D = _get_space_state()
	if not space:
		return []
	}
	var query: PhysicsShapeQueryParameters2D = PhysicsShapeQueryParameters2D.new()
	var rect: RectangleShape2D = RectangleShape2D.new()
	rect.size = Vector2(range, 16)
	query.shape = rect
	query.transform = Transform2D(0, origin + direction * range * 0.5)
	query.collision_mask = mask
	
	var results: Array[Dictionary] = space.intersect_shape(query)
	return results

## 检查指定网格位置是否可通行
## 考虑：目标方块本身 + 下方所有高方块是否阻挡
func is_passable(gx: int, gy: int, gz: int) -> bool:
	if not collision_enabled:
		return true
	}
	if not block_grid:
		return true
	}
	# 检查目标层方块
	if not block_grid.is_passable(gx, gy, gz):
		return false
	}
	# 高方块阻挡检查：下方层有 height>=2 的方块阻挡通行
	for layer: int in range(0, gz):
		var below: BlockData = block_grid.get_block(gx, gy, layer)
		if below != null and below.height >= 2:
			# 高方块会阻挡上方通行
		return false
		}
	}
	return true

## 检查角色是否可以从 (gx1,gy1,gz1) 移动到 (gx2,gy2,gz2)
func can_move(gx1: int, gy1: int, gz1: int, gx2: int, gy2: int, gz2: int) -> bool:
	return is_passable(gx2, gy2, gz2)

## 获取方块在屏幕空间的碰撞矩形
## [return] { screen_rect: Rect2, grid_pos: Vector3i, height: int }
func get_block_collision_shape(gx: int, gy: int, gz: int, height: int) -> Dictionary:
	var center: Vector2 = BlockRenderer.grid_to_screen(gx, gy, gz)
	# 菱形碰撞矩形：宽32px，高 = 8 + height*16
	var collision_height: float = TILE_HALF_H + height * TILE_SIZE
	var screen_rect: Rect2 = Rect2(
		center.x - TILE_HALF_W,
		center.y,
		TILE_HALF_W * 2,
		collision_height
	)
	return {
		"screen_rect": screen_rect,
		"grid_pos": Vector3i(gx, gy, gz),
		"height": height
	}

## 将屏幕坐标转换为网格坐标，自动检测 gz 层
## 从最高层向下扫描，找到第一个有方块（或地面）的层
func screen_to_grid_auto(screen_pos: Vector2, max_gz: int = 8) -> Vector3i:
	for gz: int in range(max_gz, -1, -1):
		var grid_pos: Vector2i = BlockRenderer.screen_to_grid(screen_pos.x, screen_pos.y, gz)
		# 该层有方块 → 命中
		if block_grid and block_grid.block_exists(grid_pos.x, grid_pos.y, gz):
			return Vector3i(grid_pos.x, grid_pos.y, gz)
		}
		# gz=0 是地面层，始终返回
		if gz == 0:
			return Vector3i(grid_pos.x, grid_pos.y, 0)
		}
	}
	return Vector3i(0, 0, 0)

	var local_pos: Vector2 = to_local(mouse_pos)
	return screen_to_grid_auto(local_pos, max_gz)

## 为不可通行的方块生成 StaticBody2D 碰撞体
func build_collision_from_grid() -> void:
	_clear_all_bodies()
	if not block_grid:
		return
	}
	var all_blocks: Dictionary = block_grid.get_all_blocks()
	for key: String in all_blocks:
		var data: BlockData = all_blocks[key] as BlockData
		if data.is_passable():
			continue  # 可通行方块不需要碰撞体
		}
		var parts: PackedStringArray = key.split(",")
		var gx: int = int(parts[0])
		var gy: int = int(parts[1])
		var gz: int = int(parts[2])
		_add_collision_body(gx, gy, gz, data)
	}

	if _collision_bodies.has(key):
		remove_collision_body(gx, gy, gz)
	}
	if data.is_passable():
		return
	}
	_add_collision_body(gx, gy, gz, data)

func remove_collision_body(gx: int, gy: int, gz: int) -> void:
	var key: String = BlockData.make_key(gx, gy, gz)
	if _collision_bodies.has(key):
		var body: StaticBody2D = _collision_bodies[key]
		if is_instance_valid(body):
			body.queue_free()
		}
		_collision_bodies.erase(key)
	}

## 网格坐标 → 屏幕坐标（菱形投影核心公式）
## [param gx] 网格 X
## [param gy] 网格 Y
## [param gz] 网格 Z（高度层），每层抬高 16px
## [return] 方块底面中心点的屏幕坐标
static func grid_to_screen(gx: int, gy: int, gz: int) -> Vector2:
	var sx: float = (gx - gy) * TILE_HALF_W
	var sy: float = (gx + gy) * TILE_HALF_H - gz * (TILE_HALF_H * 2)
	return Vector2(sx, sy)

## 屏幕坐标 → 网格坐标（逆投影，已知 gz）
## [param screen_x] 屏幕 X
## [param screen_y] 屏幕 Y
## [param gz] 假设的高度层
## [return] 最近的网格坐标
static func screen_to_grid(screen_x: float, screen_y: float, gz: int = 0) -> Vector2i:
	var offset_y: float = screen_y + gz * (TILE_HALF_H * 2)
	var gx_float: float = (screen_x / TILE_HALF_W + offset_y / TILE_HALF_H) / 2.0
	var gy_float: float = (offset_y / TILE_HALF_H - screen_x / TILE_HALF_W) / 2.0
	return Vector2i(roundi(gx_float), roundi(gy_float))

## 计算 Y-Sort 排序键
static func get_sort_key(gx: int, gy: int, gz: int) -> int:
	return (gx + gy) * Z_BASE + gz

## 为一个方块创建所有面的精灵
func _create_block_sprites(gx: int, gy: int, gz: int, data: BlockData) -> void:
	var base_pos: Vector2 = grid_to_screen(gx, gy, gz)
	var sprites: Array[Sprite2D] = []
	var key: String = BlockData.make_key(gx, gy, gz)
	var block_type: String = data.block_type

	# 1. 上平面 —— 始终渲染（菱形顶面）
	var top_tex: Texture2D = _get_texture(block_type, "top")
	var top_sprite: Sprite2D = _make_sprite(top_tex, base_pos)
	_sprite_container.add_child(top_sprite)
	sprites.append(top_sprite)

	# 2. 左垂直面 —— 放置在菱形左侧下方
	var left_tex: Texture2D = _get_texture(block_type, "left")
	# 每个高度单位渲染一段垂直面
	for h: int in range(data.height):
		var left_offset: Vector2 = Vector2(-TILE_HALF_W, TILE_HALF_H - h * TILE_SIZE)
		var left_sprite: Sprite2D = _make_sprite(left_tex, base_pos + left_offset)
		_sprite_container.add_child(left_sprite)
		sprites.append(left_sprite)
	}

	# 3. 右垂直面 —— 放置在菱形右侧下方
	var right_tex: Texture2D = _get_texture(block_type, "right")
	for h: int in range(data.height):
		var right_offset: Vector2 = Vector2(TILE_HALF_W, TILE_HALF_H - h * TILE_SIZE)
		var right_sprite: Sprite2D = _make_sprite(right_tex, base_pos + right_offset)
		_sprite_container.add_child(right_sprite)
		sprites.append(right_sprite)
	}

	# 存储精灵引用
	_sprites[key] = sprites

## 创建一个配置好的 Sprite2D
func _make_sprite(texture: Texture2D, position: Vector2) -> Sprite2D:
	var sprite: Sprite2D = Sprite2D.new()
	sprite.texture = texture
	sprite.position = position
	sprite.centered = true
	sprite.texture_filter = CanvasItem.TEXTURE_FILTER_NEAREST  # 像素风格
	return sprite

## 获取或加载纹理（带缓存）
func _get_texture(block_type: String, face: String) -> Texture2D:
	var cache_key: String = block_type + "_" + face
	if _texture_cache.has(cache_key):
		return _texture_cache[cache_key]
	}
	var path: String = PLACEHOLDER_PATH + "placeholder_block_" + block_type + "_" + face + ".png"
	if ResourceLoader.exists(path):
		var tex: Texture2D = load(path) as Texture2D
		_texture_cache[cache_key] = tex
		return tex
	}
	# 回退：使用 grass 纹理
	push_warning("BlockRenderer: 纹理未找到 " + path + "，使用 grass 回退")
	return _get_texture("grass", face)

## 移除指定位置的所有精灵

		}

## 创建一个方块面的纹理（16×16）
func _make_block_face(color: Color, face: String) -> Image:
	var img: Image = Image.create(16, 16, false, Image.FORMAT_RGBA8)
	# 填充主体颜色
	img.fill(color)
	# 外轮廓 1px 黑色边框
	var outline: Color = Color(0.1, 0.1, 0.1)
	_draw_rect_border(img, Rect2i(0, 0, 16, 16), outline)
	# 面间分隔线（上平面底部 1px）
	if face == "top":
		_draw_h_line(img, 15, _darken(color, 0.30))
	}
	return img

## 创建一个角色精灵（32×32 矩形 + 文字标识）
func _make_character_sprite(color: Color, label: String) -> Image:
	var img: Image = Image.create(32, 32, false, Image.FORMAT_RGBA8)
	img.fill(Color(0, 0, 0, 0))  # 透明背景
	# 身体矩形 (居中 24×24)
	img.fill_rect(Rect2i(4, 4, 24, 24), color)
	# 外轮廓
	_draw_rect_border(img, Rect2i(4, 4, 24, 24), Color.BLACK)
	# 文字标识（用像素模拟，在中间画简单十字）
	var text_color: Color = Color.WHITE
	# 水平线
	_draw_h_line(img, 15, text_color, 8, 24)
	# 竖直线
	_draw_v_line(img, 15, text_color, 8, 24)
	return img

## 创建一个 UI 图标精灵（16×16）
func _make_icon_sprite(color: Color) -> Image:
	var img: Image = Image.create(16, 16, false, Image.FORMAT_RGBA8)
	img.fill(Color(0, 0, 0, 0))  # 透明背景
	# 圆形图标（像素近似）
	img.fill_rect(Rect2i(2, 2, 12, 12), color)
	_draw_rect_border(img, Rect2i(2, 2, 12, 12), Color.BLACK)
	return img

## 绘制矩形边框
func _draw_rect_border(img: Image, rect: Rect2i, color: Color) -> void:
	for x: int in range(rect.position.x, rect.position.x + rect.size.x):
		img.set_pixel(x, rect.position.y, color)
		img.set_pixel(x, rect.position.y + rect.size.y - 1, color)
	}
	for y: int in range(rect.position.y, rect.position.y + rect.size.y):
		img.set_pixel(rect.position.x, y, color)
		img.set_pixel(rect.position.x + rect.size.x - 1, y, color)
	}

## 绘制水平线
func _draw_h_line(img: Image, y: int, color: Color, x_start: int = 0, x_end: int = -1) -> void:
	if x_end < 0:
		x_end = img.get_width()
	}
	for x: int in range(x_start, x_end):
		img.set_pixel(x, y, color)
	}

## 绘制竖直线
func _draw_v_line(img: Image, x: int, color: Color, y_start: int = 0, y_end: int = -1) -> void:
	if y_end < 0:
		y_end = img.get_height()
	}
	for y: int in range(y_start, y_end):
		img.set_pixel(x, y, color)
	}

## 保存 Image 为 PNG 文件
func _save_png(img: Image, filename: String) -> void:
	var path: String = OUTPUT_DIR + filename
	img.save_png(path)

## 降低颜色明度（乘以 factor）
func _darken(color: Color, factor: float) -> Color:
	return Color(color.r * factor, color.g * factor, color.b * factor, color.a)

func load_settings() -> void:
	var err: int = _config.load(CONFIG_PATH)
	if err != OK:
		# 配置文件不存在或损坏，使用默认值并保存
		_settings = DEFAULT_SETTINGS.duplicate()
		save_settings()
		return
	}
	# 从配置文件读取所有设置
	for key: String in DEFAULT_SETTINGS.keys():
		_settings[key] = _config.get_value("settings", key, DEFAULT_SETTINGS[key])
	}
	_apply_display_settings()
	_apply_audio_settings()
	settings_loaded.emit()

## 保存当前设置到文件
func save_settings() -> void:
	for key: String in _settings.keys():
		_config.set_value("settings", key, _settings[key])
	}
	_config.save(CONFIG_PATH)
	settings_saved.emit()

		return
	}
	_settings[key] = value
	setting_changed.emit(key, value)
	save_settings()

## 应用音频相关设置到 Godot 引擎
func _apply_audio_settings() -> void:
	var master_idx: int = AudioServer.get_bus_index("Master")
	if master_idx >= 0:
		AudioServer.set_bus_volume_db(master_idx, _linear_to_db(_settings["master_volume"]))
	}

## 线性值(0~1) 转 分贝值
func _linear_to_db(linear: float) -> float:
	if linear <= 0.0:
		return -80.0
	}
	return 20.0 * log(linear) / log(10.0)

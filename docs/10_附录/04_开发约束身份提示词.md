func _find_grid_under_mouse(screen_pos: Vector2) -> Vector3i:
	if not block_grid:
		return Vector3i(-999999, -999999, -999999)

	for gz: int in range(8, -1, -1):
		var gp: Vector2i = BlockRenderer.screen_to_grid(screen_pos.x, screen_pos.y, gz)
		if block_grid.has_block(gp.x, gp.y, gz):
			return Vector3i(gp.x, gp.y, gz)

	return Vector3i(-999999, -999999, -999999)


func _draw() -> void:
	if not _has_valid_hover:
		return
	if not block_grid:
		return

	var gx: int = _hovered_grid.x
	var gy: int = _hovered_grid.y
	var gz: int = _hovered_grid.z

	var block: BlockData = block_grid.get_block(gx, gy, gz)
	if block == null:
		return

	var height: int = block.height
	var center: Vector2 = BlockRenderer.grid_to_screen(gx, gy, gz)

	# ── 计算六个关键顶点 ──
	var top_v: Vector2 = center + Vector2(0, -TILE_HALF_H)
	var right_v: Vector2 = center + Vector2(TILE_HALF_W, 0)
	var left_v: Vector2 = center + Vector2(-TILE_HALF_W, 0)

	var vert_offset: float = TILE_HALF_H + height * HEIGHT_PX
	var bottom_right_v: Vector2 = center + Vector2(TILE_HALF_W, vert_offset)
	var bottom_v: Vector2 = center + Vector2(0, vert_offset)
	var bottom_left_v: Vector2 = center + Vector2(-TILE_HALF_W, vert_offset)

	# ── 绘制发光层（外宽、透明）──
	for i: int in range(GLOW_LAYERS):
		var alpha: float = GLOW_BASE_ALPHA - i * GLOW_ALPHA_DECAY
		var width: float = GLOW_BASE_WIDTH - i * GLOW_WIDTH_DECAY
		var glow_c: Color = Color(GLOW_COLOR, alpha)
		_draw_block_outline(top_v, right_v, bottom_right_v, bottom_v, bottom_left_v, left_v, glow_c, width, true)

	# ── 绘制主线（内细、不透明）──
	var main_c: Color = Color(MAIN_COLOR, MAIN_ALPHA)
	_draw_block_outline(top_v, right_v, bottom_right_v, bottom_v, bottom_left_v, left_v, main_c, MAIN_WIDTH, true)

## 三、2.5D 方块系统约束

### 3.1 核心投影公式

```
# 常量（菱形块尺寸）
TILE_HALF_W = 16   # 菱形半宽
TILE_HALF_H = 8    # 菱形半高

# 网格坐标 (gx, gy, gz) → 屏幕坐标 (sx, sy)
sx = (gx - gy) * TILE_HALF_W
sy = (gx + gy) * TILE_HALF_H - gz * (TILE_HALF_H * 2)

# 屏幕坐标 → 网格坐标（已知 gz）
gx_float = (sx / TILE_HALF_W + (sy + gz * 16) / TILE_HALF_H) / 2.0
gy_float = ((sy + gz * 16) / TILE_HALF_H - sx / TILE_HALF_W) / 2.0
# → round(gx_float), round(gy_float)
```

### 3.4 方块碰撞规则

- 空方块（`null`）可通行
- 通行类型列表：`["grass", "dirt", "sand", "farm", "plank"]`
- 碰撞形状：菱形屏幕矩形 `Rect2(center.x - 16, center.y - 8, 32, 8 + height * 16)`
- 高方块（height > 1）阻挡上方所有通行

### 4.2 场景文件规范

- 场景根节点类型 **必须反映其职责**（`CharacterBody2D` / `Node2D` / `Control` 等）

## 五、GDScript 编码标准

### 5.1 文件节顺序（严格）

```
extends               # 1. 继承声明（必须存在）
class_name            # 2. 类名（可选）
# ── 信号声明 ──
signal                # 3. 信号（如有）
# ── 常量 ──
const                 # 4. 常量
# ── 导出变量 ──
@export var           # 5. 导出变量（编辑器可调）
# ── 公开变量 ──
var                   # 6. 普通变量
# ── 私有变量 ──
var _                 # 7. 私有变量（_ 前缀）
# ── 生命周期 ──
func _ready()         # 8. _ready()
func _process(delta)  # 9. _process()（如有）
# ── 公开方法 ──
func                  # 10. 公开方法
# ── 私有方法 ──
func _                # 11. 私有方法（_ 前缀）
```

### 5.2 类型注解（强制）

```gdscript
# ✔ 正确：完整类型注解
func calculate_damage(attack: int, defense: int, multiplier: float) -> int:
	var base_damage: int = attack * multiplier
	var final_damage: int = max(0, base_damage - defense)
	return final_damage

# ✘ 错误：缺少类型注解
func calculate_damage(attack, defense, multiplier):
	var base_damage = attack * multiplier
	var final_damage = max(0, base_damage - defense)
	return final_damage
```

### 5.3 方法长度限制

- **每个方法不超过 40 行**（含空行和注释）
- 超过 40 行的方法必须拆分为多个子方法
- 子方法命名规则：`_do_subtask()`（前缀 `_` 表明内部使用）

### 5.4 私有方法约定

- 私有方法必须以 `_` 前缀命名
- 私有变量必须以 `_` 前缀命名
- 公开方法/变量无前缀

### 5.5 注释规范

```gdscript
# 单行注释：解释「为什么」而非「是什么」
var attack_power: int = 10  # 行尾注释

## 文档注释：用于公开方法/信号的说明
## [param target] 目标敌人
## [return] 实际造成的伤害值
func deal_damage(target: Node2D) -> int:
	pass

# TODO: 待实现的功能（附带责任人）
# FIXME: 已知问题（附带 Issue 编号）
# HACK: 临时解决方案（附带原因）
```

## 七、输入控制约束

### 7.1 硬性规则

```
┌────────────────────────────────────────────┐
│             输入控制硬性规则                 │
├────────────────────────────────────────────┤
│ 1. 所有输入必须通过 InputMap 动作名获取     │
│    禁止：Input.is_key_pressed(KEY_A)        │
│    禁止：Input.is_mouse_button_pressed(1)   │
│    必须：Input.is_action_just_pressed("move_left") │
│                                             │
│ 2. 输入处理集中到 PlayerController 层       │
│    禁止：在 Sprite2D 中处理 Input 事件      │
│    禁止：在 Item 脚本中处理 Input 事件      │
│    必须：PlayerController._input(event) 分发 │
│                                             │
│ 3. UI 中显示动作名而非具体键位              │
│    正确：显示「按 [跳跃] 键跳过」           │
│    错误：显示「按 Space 键跳过」            │
│    按钮图标自动适配当前输入设备             │
└────────────────────────────────────────────┘
```

### 7.2 InputMap 动作总表

**移动类**：`move_left` / `move_right` / `move_up` / `move_down` / `dodge`

**战斗类**：`attack_melee` / `draw_talisman` / `activate_seal_1~3` / `use_item` / `interact`

**符箓绘制**：`stroke_up` / `stroke_down` / `stroke_left` / `stroke_right` / `stroke_confirm`

**界面类**：`ui_inventory` / `ui_quest_log` / `ui_map` / `ui_craft` / `ui_talisman_book` / `ui_seal_script_book` / `ui_menu` / `quick_save` / `quick_load` / `screenshot`

### 7.3 输入处理流程

```
玩家操作 → InputMap 捕获（动作名）
	→ PlayerController._input(event)
		→ 状态机分发给：
			├─ MovementHandler (移动/闪避)
			├─ CombatHandler (攻击/符箓/技能)
			└─ UIHandler (菜单/交互)
```

### 7.4 控制方案

- 同时支持 **键盘 + 鼠标** 和 **手柄** 两种方案
- 手柄按钮图标在检测到手柄输入时自动切换
- 图标资源位于 `res://ui/icons/input/`

```gdscript

right_color = base_color × 0.60  # 右垂直面，明度降低 40%
outline     = base_color × 0.30  # 描边色
```

### 9.4 方块纹理绘制规则

1. **光源方向**：左上角。上平面 100%、左垂直面 80%、右垂直面 60%
2. **面间描边**：1px 深色分隔线（基准色 -30% 明度）
3. **外轮廓**：1px 黑色（`#1a1a1a`）描边
4. **纹理精度**：每个面保留完整 16×16 纹理细节
5. **垂直面平铺**：高方块垂直面沿 Y 轴 16px 无缝平铺
6. **变体规则**：每类型至少 2 变体，色值差异 ≤ 10%

## 十二、行为准则（AI 必须遵守）

### 12.1 审查前置（Review Before Writing）

1. 修改文件前，**必须完整阅读**目标文件及相关文件
2. 评估修改对以下系统的影响：
   - `SignalBus`：是否需要新增信号？
   - `GameManager`：是否需要新增全局状态？
   - `SettingsManager`：是否需要新增持久化配置？
   - 已有模块的接口是否兼容？
3. 如修改涉及架构变更，必须先在架构文档中同步更新

### 12.2 不确定时询问（Ask When Uncertain）

以下情况**必须确认后再执行**：

| 情况 | 举例 |
|:-----|:-----|
| 功能边界不清晰 | "添加攻击系统"未说明近战还是远程 |
| 多种实现方案有取舍 | 列出选项让用户选择 |
| 涉及架构修改 | 新增 Autoload、修改 SignalBus 信号签名 |
| 资源命名/路径不确定 | 新增资源路径 |
| 用户表述模糊 | "优化它"、"让它好看点" |
| 出现疑问或不明确的点 | 反复询问直到完全确定后再执行 |

### 12.3 分阶段交付（Deliver in Phases）

1. 复杂需求拆分为 **5～10 个阶段**
2. 每个阶段完成后**暂停，等待用户确认**
3. 确认后再进入下一阶段
4. 每个阶段产出可验证的成果

### 12.4 严格访问控制（Strict Access Control）

1. **不修改任何未经用户同意的文件**
2. 修改前**必须先展示修改计划**
3. 用户确认后方可执行
4. 即使修复明显 bug，也需先说明

### 12.5 变更后汇总（Post-Change Summary）

每次文件修改后，必须列出**所有变更文件**的完整路径：

```
变更文件：
- [`scenes/player/player.tscn`](scenes/player/player.tscn) — 新增攻击状态节点
- [`scripts/player/player.gd`](scripts/player/player.gd) — 添加攻击状态机逻辑
- [`scripts/autoload/signal_bus.gd`](scripts/autoload/signal_bus.gd) — 新增 player_attacked 信号
```

## 十四、Godot 版本兼容性与引擎约束

### 14.1 引擎版本

- 目标引擎：**Godot 4.3+**
- 渲染器：Forward+（默认）
- 使用 GDScript 1.0（Godot 4 原生）

### 14.2 禁止使用的功能

| 功能 | 原因 |
|:-----|:-----|
| `C#` 脚本 | 项目统一使用 GDScript |
| GDScript 2.0 静态类型以外的动态特性 | 类型安全 |
| `EditorPlugin` / `EditorScript` | 运行时不需要编辑器功能 |
| `Thread` / `Mutex`（除非 BlockController 性能优化）| 线程安全问题 |
| `@rpc` 注解 | 无多人需求 |
| `NavigationServer2D` | 使用方块网格寻路替代 |

### 14.3 推荐使用的模式

| 模式 | 说明 |
|:-----|:-----|
| `Resource` 子类 | 所有数据定义（装备/怪物/成就/符箓/配方）|
| `@export` + `@onready` | 节点引用和资源依赖 |
| `Signal` | 模块间所有通信 |
| `class_name` | 所有可复用类型 |
| `@tool` 脚本 | 仅在编辑器工具脚本中使用 |
